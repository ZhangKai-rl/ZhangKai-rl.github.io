将多个区间，可能有重复的进行合并。
经典题目：给定n个区间，合并所有有交集的区间

有一维和二维区间合并，二维看格子染色。

# 模板

```cpp
#inlcude <bits/stdc++.h>
using namespace std;
typedef pair<int, int> PII;
vector<PII> segs;

void merge(vector<PII> &segs) {
	vector<PII> ans;

	sort(segs.begin(), segs.end());

	int st = -2e9, ed = -2e9;
	for (auto seg : segs) {
		if (ed < seg.first) {
			if (st != -2e9) ans.push_back({st, ed});
			st = seg.first, ed = seg.second;
		} else {
			ed = max(ed, seg.second);
		}
	}
	if (st != -2e9) ans.push_back({st, ed});
	segs = ans;
}

int main() {
	int n;
	cin >> n;
	for (int i = 0; i < n; i ++) {
		int l, r;
		cin >> l >> r;
		segs.push_back({l, r});
	}
	merge(segs);
	cout << segs.size() << endl;
}
```

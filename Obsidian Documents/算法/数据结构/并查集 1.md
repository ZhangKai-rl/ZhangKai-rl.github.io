是一种**树形**数据结构。
包含两种操作：
- 合并unionset
- 查找find

# 模板


```cpp
fa[N];

int find(int x) {
	if (fa[x] == x) return x;
	return fa[x] = find(fa[x]); // 进行了路径压缩
}
void unionset(int x, int y) {
	fa[find(x)] = find(y);
}
vector<int> siz(N, 1);
void unionset(int x, int y) {
	x = find(x), y = find(y);
	if (x == y) return ;
	if (siz[x] > siz[y]) swap(x, y);
	fa[x] = y;
	siz[y] += siz[x];
}

int main() {
	cin >> n >> m;
	for (int i = 1; i <= n; i ++) fa[i] = i;
	while (m -- ) {
		cin >> z >> x >> y;
		if (z == 1) unionset(x, y);
		else {
			x = find(x), y = find(y);
			if (x == y) puts("y");
			else puts("n");
		}
	}
}
```
